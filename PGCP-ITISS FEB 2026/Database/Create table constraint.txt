Create table student
(sid int primary key,
sname varchar(25)  not null,
city varchar(20),
age  int check(age>21),
pincode int default 0
);



Create table student1
(sid int,
sname varchar(25) not null,
city varchar(20),
age  int ,
pincode int default 0,
CONSTRAINT PK_student1 PRIMARY KEY (sid),
CONSTRAINT chk_student1 CHECK (Age >= 18) 
);


 create table course
 (cid int,
  cname varchar(30) not null,
  duration int,
  primary key (cid)
  );


 create table student3
  (
  sid int primary key,
  sname varchar(30) not null,
  courseid int,
  constraint st3_fk foreign key (courseid) references course(cid)
 );
